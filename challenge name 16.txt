#include <stdio.h>

int main() {
    int n, i;
    int sum = 0;

    scanf("%d", &n);

    for (i = 1; i <= n / 2; i++) {
        if (n % i == 0) {
            sum = sum + i;
        }
    }

    if (sum == n) {
        printf("Yes");
    } else {
        printf("No");
    }

    return 0;
}
