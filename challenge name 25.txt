#include <stdio.h>
int main() {
    int n, temp, digit;
    int count = 0;
    int sum = 0;
    int power, i;

    scanf("%d", &n);

    temp = n;


    for (; temp > 0; temp = temp / 10) {
        count++;
    }

    temp = n;


    for (; temp > 0; temp = temp / 10) {
        digit = temp % 10;
        power = 1;

        for (i = 1; i <= count; i++) {
            power = power * digit;
        }

        sum = sum + power;
    }

    if (sum == n) {
        printf("Yes");
    } else {
        printf("No");
    }

    return 0;
}
