#include <stdio.h>

int main() {
    int n, temp, digit;
    int sum = 0;
    int fact, i;

    scanf("%d", &n);

    temp = n;

    for (; temp > 0; temp = temp / 10) {
        digit = temp % 10;
        fact = 1;

        for (i = 1; i <= digit; i++) {
            fact = fact * i;
        }

        sum = sum + fact;
    }

    if (sum == n) {
        printf("Yes");
    } else {
        printf("No");
    }

    return 0;
}
